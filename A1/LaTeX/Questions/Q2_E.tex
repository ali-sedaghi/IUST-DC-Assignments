\SubProblem
{ه}
{
    با توجه به اینکه این مسئله دارای دو متغیر تصادفی 
    \lr{X}
    و
    \lr{Y}
    است، رابطه فوق به شکل زیر درمی‌آید:
    
    \begin{equation*}
        H(X, Y) \leq H(X) + H(Y)
    \end{equation*}
    
    همانطور که در ذیل مشاهده می‌کنید این رابطه برقرار است.
    
    \begin{equation*}
        \frac{27}{8} \le \frac{7}{4} + 2 \Rightarrow \frac{27}{8} \le \frac{15}{4}
    \end{equation*}
}
